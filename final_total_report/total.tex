\section{引言}

二维无穷网格电路的输运性质是凝聚态物理、电路理论与工程应用中的重要问题。
电阻网格的等效电阻计算不仅是电学中的经典问题，其在网络分析
和数值算法中亦有重要应用。与此同时，由电容和电感构成的$LC$网格则展现了
更为丰富的频率响应特性，与离散微波、超材料密切相关。

离散Fourier变换（Discrete Fourier Transform, DFT）是处理此类离散平移对称
系统的有力工具。通过将实空间中的差分方程变换至频域，可将复杂的求和或积分
问题转化为代数运算，再通过逆变换还原物理量。本文从纯电阻网格入手，
通过展示DFT在纯电阻网格中的性质，从而以相似的方式迁移运用到LC网格中。
其中还介绍了$LC$电路构造想法的由来——一维SSH电路，受到其中阻抗为0时电路
奇异性质的启发，本文探究了其$Z(\omega)=0$的特殊情况，发现横纵坐标奇偶性相同的点
与原点之间的阻抗均为0，其余点阻抗均为定值的现象，并对其展开进一步研究。


\section{纯电阻网格的Fourier分析}

\subsection{问题设定与基尔霍夫方程}

考虑一个二维无穷电阻网格，每个格点通过电阻$r$与上下左右四个相邻格点相连。
在原点$(0,0)$注入电流$I$，在节点$(m,n)$流出电流$-I$。任意节点$(x,y)$的
电势记为$V(x,y)$。源电流分布为
\begin{equation}
  I(x,y)=\begin{dcases}
    I,&x=y=0,\\
    -I,&x=m,\ y=n,\\
    0,&\text{其他}.
  \end{dcases}
\end{equation}

根据基尔霍夫电流定律（KCL），对任意节点$(x,y)$，流出电流之和等于该点注入电流：
\begin{align}
  I(x,y)&=\frac{1}{r}\left[V(x,y)-V(x-1,y)\right]
        +\frac{1}{r}\left[V(x,y)-V(x+1,y)\right]\nonumber\\
        &+\frac{1}{r}\left[V(x,y)-V(x,y-1)\right]
        +\frac{1}{r}\left[V(x,y)-V(x,y+1)\right].
\end{align}

定义离散拉普拉斯算子
\begin{equation}
  \mathscr{L}V(x,y)=\frac{1}{4}\left[V(x-1,y)+V(x+1,y)
  +V(x,y-1)+V(x,y+1)\right],
\end{equation}
则KCL方程可写为
\begin{equation}
  V(x,y)-\mathcal{L}V(x,y)=\frac{r}{4}I(x,y).
\end{equation}

待求的等效电阻为
\begin{equation}
  R(m,n)=\frac{1}{I}\left[V(0,0)-V(m,n)\right].
\end{equation}

\subsection{Fourier变换与频域求解}

引入二维离散Fourier变换对
\begin{equation}
  F(\omega_1,\omega_2)=\fo{V(x,y)}
  =\sum_{x=-\infty}^{\infty}\sum_{y=-\infty}^{\infty}V(x,y)\e{\ii(x\omega_1+y\omega_2)},
\end{equation}
其逆变换为
\begin{equation}
  V(x,y)=\f{F(\omega_1,\omega_2)}
  =\frac{1}{4\pi^2}\int_{-\pi}^{\pi}\int_{-\pi}^{\pi}
  F(\omega_1,\omega_2)\e{-\ii(x\omega_1+y\omega_2)}\,\di\omega_1\di\omega_2.
\end{equation}

对源电流分布进行Fourier变换：
\begin{align}
  \fo{I(x,y)}
  &=\sum_{x,y}I(x,y)\e{\ii(x\omega_1+y\omega_2)}
  =I\left[1-\e{\ii(m\omega_1+n\omega_2)}\right].
\end{align}

利用Fourier变换的频移性质，可得
\begin{equation}
  \fo{\mathcal{L}V(x,y)}
  =F(\omega_1,\omega_2)\frac{\cos\omega_1+\cos\omega_2}{2}.
\end{equation}

将方程(4)两端作Fourier逆变换，得到频域电势
\begin{equation}
  F(\omega_1,\omega_2)=\frac{Ir}{2}\,
  \frac{1-\e{\ii(m\omega_1+n\omega_2)}}{2-(\cos\omega_1+\cos\omega_2)}.
\end{equation}

于是电势的实空间表达式为
\begin{align}
  V(x,y)&=\frac{Ir}{8\pi^2}
  \iint_{\Omega}\frac{\left[1-\e{\ii(m\omega_1+n\omega_2)}\right]
  \e{-\ii(x\omega_1+y\omega_2)}}
  {2-(\cos\omega_1+\cos\omega_2)}\,\di\omega_1\di\omega_2,
\end{align}
其中$\Omega=[-\pi,\pi]^2$。

考虑到积分区域的对称性，虚部消失，可得
\begin{align}
  V(0,0)&=\frac{Ir}{8\pi^2}
  \iint_{\Omega}\frac{1-\cos(m\omega_1+n\omega_2)}
  {2-(\cos\omega_1+\cos\omega_2)}\,\di\omega_1\di\omega_2,\\
  V(m,n)&=-\frac{Ir}{8\pi^2}
  \iint_{\Omega}\frac{1-\cos(m\omega_1+n\omega_2)}
  {2-(\cos\omega_1+\cos\omega_2)}\,\di\omega_1\di\omega_2.
\end{align}

因此等效电阻的积分解为
\begin{equation}
  R_{mn}=\frac{r}{4\pi^2}
  \iint_{\Omega}\frac{1-\cos(m\omega_1+n\omega_2)}
  {2-(\cos\omega_1+\cos\omega_2)}\,\di\omega_1\di\omega_2.
\label{eq:R_integral}
\end{equation}

\subsection{积分向级数的转化}

积分式(\ref{eq:R_integral})虽形式简洁，但直接数值计算二维奇异积分较为不便。
下面将其化为快速收敛的级数形式。

记
\begin{equation}
  I(m,n)=\iint_{\Omega}\frac{1-\cos(m\omega_1+n\omega_2)}
  {2-(\cos\omega_1+\cos\omega_2)}\,\di\omega_1\di\omega_2.
\end{equation}

作变量变换
\begin{equation}
  u=\frac{\omega_1+\omega_2}{2},\qquad v=\frac{\omega_1-\omega_2}{2},
\end{equation}
则
\begin{equation}
  \di\omega_1\di\omega_2=2\,\di u\di v,
\end{equation}
积分区域变为$\Omega'=\{(u,v): |u|+|v|\leqslant \pi\}$。

令$a=m+n,\ b=m-n$，积分化为
\begin{align}
  I(a,b)&=\iint_{\Omega'}\frac{1-\cos(au+bv)}{1-\cos u\cos v}\,\di u\di v\nonumber\\
  &=\iint_{\Omega'}(1-\cos(au+bv))
  \sum_{k=0}^{\infty}\cos^k u\cos^k v\,\di u\di v\nonumber\\
  &=I_1-I_2,
\end{align}
其中
\begin{align}
  I_1&=\sum_{k=0}^{\infty}\iint_{\Omega'}\cos^k u\cos^k v\,\di u\di v,\\
  I_2&=\sum_{k=0}^{\infty}\iint_{\Omega'}\cos(au+bv)\cos^k u\cos^k v\,\di u\di v.
\end{align}

首先计算$I_1$。利用对称性，
\begin{align}
  I_1&=2\sum_{k=0}^{\infty}\int_0^\pi\int_0^\pi \cos^k u\cos^k v\,\di u\di v\nonumber\\
  &=2\sum_{k=0}^{\infty}\left(\int_0^\pi \cos^k u\,\di u\right)^2\nonumber\\
  &=2\pi^2\sum_{k=0}^{\infty}\left(\frac{(2k-1)!!}{(2k)!!}\right)^2
  =2\pi^2\sum_{k=0}^{\infty}\frac{1}{2^{4k}}\frac{(2k)!^2}{k!^4}.
\end{align}

其次计算$I_2$。由于被积函数关于原点对称，
\begin{align}
  I_2&=2\sum_{k=0}^{\infty}
  \left(\int_0^\pi \cos(au)\cos^k u\,\di u\right)
  \left(\int_0^\pi \cos(bv)\cos^k v\,\di v\right).
\end{align}

定义辅助积分
\begin{equation}
  J(a)=\frac{1}{2}\int_{-\pi}^{\pi}\cos(au)\cos^k u\,\di u.
\end{equation}

令$z=\e{\ii u}$，则$\di u=\di z/(\ii z)$，留数计算(计算过程见附录\ref{A})可得
\begin{align}
  J(a)&=\frac{\pi}{2^k}\,
  \frac{k!}{\left(\frac{k+a}{2}\right)!\left(\frac{k-a}{2}\right)!},
  \quad |a|\leqslant k,\ k\equiv a\pmod 2,
\end{align}
其他情况$J(a)=0$。

同理
\begin{equation}
  J(b)=
  \begin{dcases}
    \frac{\pi}{2^k}\,
    \frac{k!}{\left(\frac{k+b}{2}\right)!\left(\frac{k-b}{2}\right)!},
    & |b|\leqslant k,\ k\equiv b\pmod 2,\\
    0, & \text{其他}.
  \end{dcases}
\end{equation}

非零贡献要求$a,b$同奇偶，且$|a|\geqslant |b|$。令$k=a+2p$（$p=0,1,2,\dots$），
可得
\begin{align}
  I_2=\frac{2\pi^2}{2^{2(m+n)}}\sum_{k=0}^{\infty}\frac{1}{2^{4k}}
  \frac{(m+n+2k)!}{k!(m+k)!(n+k)!(m+n+k)!}.
\end{align}

最终，等效电阻的级数公式为
\begin{equation}
  R_{mn}=\frac{r}{2}\sum_{k=0}^{\infty}\frac{1}{2^{4k}}
  \Bigg[\left(\frac{(2k)!}{k!k!}\right)^2\nonumber-\frac{(m+n+2k)!^2}
  {2^{2(m+n)}k!(m+k)!(n+k)!(m+n+k)!}\Bigg].
\label{eq:R_series}
\end{equation}

公式(\ref{eq:R_series})收敛迅速，适合数值计算。当$m=n=0$时，$R_{00}=0$；
对于$m,n$较大时，级数项以$2^{-4k}$衰减。

\subsection{各点电压值理论推导}
晶胞编号：$1\leqslant i \leqslant N$，节点编号：$A,B$。考虑第$n$和$n+1$个晶胞中的节点，其电压分别为$V_{A}^n$和$V_{B}^n$。
根据KCL，得到以下方程组：
\begin{equation}
  \begin{dcases}
    \ii\omega C_1\left(V_{A}^n - V_{B}^n\right) + \ii\omega C_2\left(V_{A}^n - V_{B}^{n-1}\right)+ \frac{V_{A}^n-0}{\ii\omega L} = 0,\\
    \ii\omega C_1\left(V_{B}^{n-1} - V_{A}^{n-1}\right) + \ii\omega C_2\left(V_{B}^{n-1} - V_{A}^{n}\right)+ \frac{V_{B}^{n-1}-0}{\ii\omega L} = 0. 
  \end{dcases}
\end{equation}

定义
\begin{equation}
  V[n]=\begin{bmatrix}
    V_A^n\\
    V_B^n
  \end{bmatrix}
\end{equation}

于是可以得到

\begin{equation}
  V[n]=\begin{bmatrix}
-\frac{C_1}{C_2} & 
\frac{1}{C_2} \left[(C_1 + C_2) - \frac{1}{\omega^2 L} \right] \\
-\frac{1}{C_2} \left[(C_1 + C_2) - \frac{1}{\omega^2 L} \right] & 
- \frac{C_2}{C_1}+\frac{1}{C_1 C_2}\left[(C_1 + C_2) - \frac{1}{\omega^2 L} \right]^2 

  \end{bmatrix}V[n-1]
\end{equation}
其中$n=1,2,\dots,N-1$。













\section{$LC$网格的阻抗积分}

\subsection{问题背景与频域方程}

现在我们考虑另一类二维无穷网格：横向（$x$方向）由电容$C$连接，纵向（$y$方向）
由电感$L$连接。在节点$(0,0)$注入交流电流$I$（角频率$\omega$），在$(m,n)$流出。
任意节点的电势记为$V(x,y)$。

根据KCL，对任意节点$(x,y)$，流出电流之和为
\begin{align}
  &\frac{V(x,y)-V(x-1,y)}{1/(\ii\omega C)}
  +\frac{V(x,y)-V(x+1,y)}{1/(\ii\omega C)}\nonumber\\
  &+\frac{V(x,y)-V(x,y-1)}{\ii\omega L}
  +\frac{V(x,y)-V(x,y+1)}{\ii\omega L}
  =I(x,y),
\end{align}
其中$I(x,y)$的定义与式(1)相同。

整理并乘以$-\ii$，得到
\begin{align}
  &\omega C\left[2V(x,y)-V(x-1,y)-V(x+1,y)\right]\nonumber\\
  &-\frac{1}{\omega L}\left[2V(x,y)-V(x,y-1)-V(x,y+1)\right]
  =-\ii I(x,y).
\label{eq:LC_KCL}
\end{align}

采用与上节相同的Fourier变换约定。在频域中，差分算子化为
\begin{align}
  2V(x,y)-V(x-1,y)-V(x+1,y)
  &\longrightarrow 4\sin^2\frac{k_1}{2}\,\tilde V(k_1,k_2),\\
  2V(x,y)-V(x,y-1)-V(x,y+1)
  &\longrightarrow 4\sin^2\frac{k_2}{2}\,\tilde V(k_1,k_2).
\end{align}

对源电流作变换：
\begin{equation}
  \tilde I(k_1,k_2)=I\left(1-\e{\ii(mk_1+nk_2)}\right).
\end{equation}

将上述结果代入方程(\ref{eq:LC_KCL})，得到频域电势：
\begin{align}
  \tilde V(k_1,k_2)=
  \frac{\omega L I\left(1-\e{\ii(mk_1+nk_2)}\right)}
  {4\ii\left(\omega^2 LC\sin^2\frac{k_1}{2}-\sin^2\frac{k_2}{2}\right)}.
\label{eq:LC_V_freq}
\end{align}

\subsection{阻抗积分表达式与格林函数}

任意两点$(m,n)$与$(0,0)$之间的等效阻抗定义为
\begin{equation}
  \tilde Z(\omega)=\frac{V(m,n)-V(0,0)}{I}.
\end{equation}

利用Fourier逆变换将式(\ref{eq:LC_V_freq})代入，可得阻抗的双重积分表达式：
\begin{align}
  \tilde Z(\omega)=
  \frac{-\ii\omega L}{4\pi^2}
  \int_{-\pi}^{\pi}\int_{-\pi}^{\pi}
  \frac{1-\e{\ii(mk_1+nk_2)}}
  {\omega^2 LC\sin^2\frac{k_1}{2}-\sin^2\frac{k_2}{2}}
  \,\di k_1\di k_2.
\label{eq:Z_integral_2D}
\end{align}

引入核心的二维晶格格林函数
\begin{equation}
    I(m,n;\alpha) \equiv
    \int_{-\pi}^{\pi}\int_{-\pi}^{\pi}
    \frac{1-\cos(mk_1)\cos(nk_2)}
    {\alpha\sin^2\frac{k_1}{2}-\sin^2\frac{k_2}{2}}
    \,\di k_2\,\di k_1,
    \label{eq:I_mn_definition}
\end{equation}
其中 $\alpha \equiv \omega^2 LC$ 是无量纲频率参数。式(\ref{eq:Z_integral_2D})中的虚部在积分后贡献为零，因此有
\begin{equation}
    \tilde Z(\omega) = \frac{-\ii\omega L}{4\pi^2} I(m,n;\alpha).
    \label{eq:Z_I_relation}
\end{equation}
因此，$I(m,n;\alpha)$ 的解析性质直接决定了阻抗的行为。

利用半角公式 $\sin^2(x/2) = (1-\cos x)/2$，分母可改写为
\begin{equation}
    \alpha \sin^2\frac{k_1}{2} - \sin^2\frac{k_2}{2}
    = \frac{B(k_1) + \cos k_2}{2},
    \label{eq:denominator_transform}
\end{equation}
其中
\begin{equation}
    B(k_1) \equiv \alpha(1-\cos k_1) - 1.
    \label{eq:B_definition}
\end{equation}

对内层 $k_2$ 积分进行解析处理。定义
\begin{equation}
    J_n(B) = \int_{-\pi}^{\pi} \frac{\cos(n\theta)}{B + \cos\theta} \di\theta, \quad n=0,1,2,\dots
    \label{eq:J_n_definition}
\end{equation}
利用 $\cos(mk_1)\cos(nk_2) = \cos(mk_1)\cos(n\theta)$，原积分化为
\begin{equation}
    I(m,n;\alpha) = 2\int_{-\pi}^{\pi} \left[J_0(B) - \cos(mk_1) J_n(B)\right] \di k_1.
    \label{eq:I_to_J}
\end{equation}

\subsection{内层积分 $J_n(B)$ 的闭式}

积分 $J_n(B)$ 的解析闭式依赖于参数 $B$ 的取值（详细推导见附录A）。结果如下：

\begin{enumerate}
    \item 当 $|B| > 1$ 时，分母无实零点，积分正则：
    \begin{equation}
        J_n(B) = \frac{2\pi z_{\text{in}}^n}{\sqrt{B^2-1}} \operatorname{sgn}(B),
        \label{eq:J_n_regular}
    \end{equation}
    其中 $z_{\text{in}}$ 是方程 $z^2+2Bz+1=0$ 位于单位圆内的根。
    \item 当 $|B| \le 1$ 时，积分理解为柯西主值：
    \begin{equation}
        J_0(B) = 0, \quad J_n(B) = 2\pi U_{n-1}(-B), \quad n\ge 1,
        \label{eq:J_n_PV}
    \end{equation}
    其中 $U_k(x)$ 是第二类 Chebyshev 多项式。
\end{enumerate}

这一区分至关重要，因为它直接导致了 $I(m,n;\alpha)$ 在不同 $\alpha$ 区间的截然不同的行为。

\subsection{$\alpha < 1$ 时的严格恒零条件}

当 $\alpha < 1$ 时，由 $B = \alpha(1-\cos k_1) - 1$，可知 $B \in [-1, 2\alpha-1] \subset [-1, 1)$，即对所有 $k_1$ 都有 $|B| \le 1$。因此全域适用主值公式 (\ref{eq:J_n_PV})。代入式 (\ref{eq:I_to_J}) 得
\begin{equation}
    I(m,n;\alpha) = -4\pi \int_{-\pi}^{\pi} \cos(mk_1) U_{n-1}(1-\alpha+\alpha\cos k_1) \di k_1.
    \label{eq:I_alpha_less_1}
\end{equation}

$U_{n-1}(x)$ 是 $x$ 的 $n-1$ 次多项式。复合 $x=1-\alpha+\alpha\cos k_1$ 后，被积函数是 $\cos k_1$ 的至多 $n-1$ 次多项式。由傅里叶展开，$\cos^j k_1$ 可展开为 $\{\cos(\ell k_1): \ell=0,1,\dots,j\}$ 的线性组合（见附录D）。因此，当 $m \ge n$ 时，$\cos(mk_1)$ 的频率 $m$ 严格大于展开式中出现的所有余弦频率（均 $\le n-1$），由傅里叶正交性，每一项积分为零。从而得到核心结论：
\begin{equation}
    \boxed{\alpha < 1,\ m \ge n \implies I(m,n;\alpha) \equiv 0.}
    \label{eq:theorem_alpha_less_1}
\end{equation}
这是一个恒零性质，而非在特定 $\alpha$ 处的孤立零点。当 $m < n$ 时，积分结果为 $\alpha$ 的多项式（次数 $n-1$），典型例子如
\begin{equation}
    I(0,1) = -8\pi^2, \quad I(0,2) = 16\pi^2(\alpha-1), \quad I(1,2) = -8\pi^2\alpha, \quad I(1,3) = 32\pi^2\alpha(\alpha-1).
    \label{eq:I_examples}
\end{equation}

\subsection{$\alpha = 1$ 的精确解}

在临界点 $\alpha=1$，$B = -\cos k_1$，$-B = \cos k_1$。利用 Chebyshev 多项式的三角恒等式 $U_{n-1}(\cos k_1) = \frac{\sin(nk_1)}{\sin k_1}$（见附录B），式 (\ref{eq:I_alpha_less_1}) 变为
\begin{equation}
    I(m,n;1) = -4\pi \int_{-\pi}^{\pi} \cos(mk_1) \frac{\sin(nk_1)}{\sin k_1} \di k_1.
    \label{eq:I_alpha_eq_1_start}
\end{equation}
利用积化和差和引理 $\int_{-\pi}^{\pi} \frac{\sin(K\theta)}{\sin\theta} \di\theta = 2\pi$（$K$ 为奇数），$0$（$K$ 为偶数）（见附录C），可得
\begin{equation}
    \boxed{
    I(m,n;1) =
    \begin{cases}
        -8\pi^2, & m < n \ \text{且}\ m,n \ \text{一奇一偶},\\
        0, & \text{其他情况}.
    \end{cases}
    }
    \label{eq:theorem_alpha_eq_1}
\end{equation}

\subsection{$\alpha > 1$ 的讨论}

当 $\alpha > 1$ 时，$B \in [-1, 2\alpha-1]$，积分域分裂为：
\begin{itemize}
    \item 主值区 $(|B| \le 1)$：$|\cos k_1| \ge 1 - 2/\alpha$；
    \item 正则区 $(B > 1)$：$|\cos k_1| < 1 - 2/\alpha$。
\end{itemize}

对于 $n=0$ 的特殊情况，$J_0(B)$ 在主值区为零，在正则区为正。被积函数 $2J_0(B)(1-\cos(mk_1)) \ge 0$ 且不恒为零，因此
\begin{equation}
    \forall \alpha > 1, \forall m > 0: I(m,0;\alpha) > 0.
    \label{eq:theorem_alpha_greater_1_n0}
\end{equation}
对于一般 $m,n > 0$，阻抗行为更为复杂，$m\ge n$ 时通常保号无零点，$m<n$ 时可能因两区贡献干涉而产生零点。这些细节可通过数值计算进一步研究。







\section{数值实验与讨论}

根据上节的严格理论，$\tilde Z(\omega)$ 的行为可以精确刻画。

\begin{enumerate}
    \item 当 $\alpha = \omega^2 LC < 1$（即 $\omega < 1/\sqrt{LC}$）时：
    \begin{itemize}
        \item 若 $m \ge n$，则 $I(m,n;\alpha) \equiv 0$，因此 $\tilde Z(\omega) \equiv 0$。
        \item 若 $m < n$，则 $I(m,n;\alpha)$ 是 $\alpha$ 的多项式，阻抗为纯虚数，且在该区间内无零点。
    \end{itemize}
    
    \item 当 $\alpha = 1$（即 $\omega = 1/\sqrt{LC}$）时：
    \begin{itemize}
        \item 若 $m < n$ 且 $m,n$ 一奇一偶，则 $I(m,n;1) = -8\pi^2$，阻抗为
        \begin{equation}
            \tilde Z = \frac{-\ii\omega L}{4\pi^2}(-8\pi^2) = 2\ii\omega L = 2\ii\sqrt{\frac{L}{C}}.
        \end{equation}
        \item 其他情况，阻抗为零。
    \end{itemize}
    
    \item 当 $\alpha > 1$（即 $\omega > 1/\sqrt{LC}$）时，阻抗一般不为零，且其行为取决于 $m,n$ 的具体值。
\end{enumerate}

图\ref{fig:Z_omega}展示了 $m=1,n=0$ 时阻抗模值 $|\tilde Z(\omega)|$ 随频率的变化。根据理论，该情况下 $\tilde Z(\omega)$ 在 $\omega < 1/\sqrt{LC}$ 时恒为零，在 $\omega = 1/\sqrt{LC}$ 处为零，在 $\omega > 1/\sqrt{LC}$ 时随频率增大而增大。这清晰地展示了带隙和通带行为。

\section{结论}

本文运用离散Fourier变换系统研究了两类二维无穷网格电路的输运性质，主要成果如下：

\begin{enumerate}
  \item 对于纯电阻网格，从基尔霍夫定律出发，通过Fourier变换获得电势的频域解，
        进而导出等效电阻的双重积分表达式。进一步借助变量替换和留数技巧，
        将积分化为快速收敛的双重级数（式(\ref{eq:R_series})），
        避免了二维奇异积分的直接计算。
  \item 对于$LC$网格，建立了频域阻抗积分模型。通过引入二维晶格格林函数 $I(m,n;\alpha)$ 并利用留数定理对内层积分进行严格解析，首次给出了该积分在所有 $\alpha$ 区间的完整解析行为。核心结论包括：
        \begin{itemize}
            \item 当 $\alpha < 1$ 且 $m \ge n$ 时，$I(m,n;\alpha) \equiv 0$，揭示了阻抗的严格零区间；
            \item 在临界点 $\alpha=1$，给出了 $I(m,n;1)$ 的精确表达式，修正了文献中关于特定 $\alpha$ 处“零点”的错误论断；
            \item 对于 $\alpha > 1$，给出了定性行为描述。
        \end{itemize}
  \item 两种问题的对比表明，Fourier变换是处理离散平移对称系统的一般性方法，
        而后续的积分化简策略（级数展开或留数降维）则依赖于具体算子的类型
        （椭圆型或双曲型）。
\end{enumerate}

本文的推导过程完整、自洽，所有中间步骤均有详细交代，既适合作为电动力学或
数学物理方法课程的教学补充材料，也可为相关工程计算提供理论参考。

\section{$I$的计算过程}\label{A}
记
\begin{equation}
  J(a)=\dfrac{1}{2}\int_{-\pi}^\pi\cos au\cos^ku\di u
\end{equation}

令$z=\e{\ii u}$，则有
\begin{equation}
  \di z=\ii\e{\ii u}\di u=\ii z\di u
\end{equation}
\begin{align*}
  J(a)&=\dfrac{1}{2\ii}\oint_{|z|\leqslant1}\dfrac{z^a+z^{-a}}{2}\dfrac{\left(z+z^{-1}\right)^k}{2^k}z^{-1}\di z\\
  &=\dfrac{1}{2^{k+2}\ii}\oint_{|z|\leqslant1}\dfrac{\left(z^{2a}+1\right)}{z^{a+k+1}}\left(z^2+1\right)^k\di z\\
  &=\dfrac{1}{2^{k+2}\ii}\oint_{|z|\leqslant1}\sum_{t=0}^{k}\binom{k}{t}z^{2t}\dfrac{\left(z^{2a}+1\right)}{z^{a+k+1}}\di z\\
  &=\dfrac{-\ii}{2^{k+2}}\sum_{t=0}^{k}\binom{k}{t}\oint_{|z|\leqslant1}\left(\dfrac{1}{z^{k+a+1-2t}}+\dfrac{1}{z^{k-a+1-2t}}\right)\di z\\
\end{align*}

利用留数定理，只需算被积函数洛朗展开$z^{-1}$系数
\begin{align*}
  J(a)&=\dfrac{\pi}{2^{k+1}}\left[\binom{k}{\frac{k+a}{2}}+\binom{k}{\frac{k-a}{2}}\right]\\
  &=\dfrac{\pi}{2^k}\dfrac{k!}{\left(\frac{k+a}{2}\right)!\left(\frac{k-a}{2}\right)!}
\end{align*}

考虑$J(a)$关于$k$的取值，
\begin{equation}
  J(a)=
  \begin{dcases}
    \dfrac{\pi}{2^k}\dfrac{k!}{\left(\frac{k+a}{2}\right)!\left(\frac{k-a}{2}\right)!},&|a|\leqslant k,k\equiv a(\text{mod}2)\\
    0,&\text{other}.
  \end{dcases}
\end{equation}

同理于$J(a)$，
\begin{equation}
  J(b)=
  \begin{dcases}
    \dfrac{\pi}{2^k}\dfrac{k!}{\left(\frac{k+b}{2}\right)!\left(\frac{k-b}{2}\right)!},&|b|\leqslant k,k\equiv b(\text{mod}2)\\
    0,&\text{other}.
  \end{dcases}
\end{equation}

考虑$J(a)J(b)$：
\begin{enumerate}
  \item $a,b$必然同奇偶；
  \item 三角不等式$|a|\geqslant|b|$。
\end{enumerate}
故$I_2$的求和中只有$k=a+2p,p=0,1,2,\cdots$对求和有贡献。所以，$I_2$中关于$k$的求和换成对$p$的求和
\begin{align}
  I_2=\sum_{p=0}^{\infty}\dfrac{2\pi^2}{2^{2a+4p}}\dfrac{(a+2p)!}{p!(a+p)!}\dfrac{(a+2p)!}{(\frac{a-b}{2}+p)!(\frac{a+b}{2}+p)!}
\end{align}

注意$a,b$的定义式，并将$p$替换回$k$，又有
\begin{equation}
  I_2=\dfrac{2\pi^2}{2^{2(m+n)}}\sum_{k=0}^{\infty}\dfrac{1}{2^{4k}}\dfrac{(m+n+2k)!}{k!(m+k)!(n+k)!(m+n+k)!}
\end{equation}

结合$I_1,I_2$的表达式，得出积分$I$的级数为
\begin{align*}
  I&=I_1-I_2\\
  &=2\pi^2\sum_{k=0}^{\infty}\dfrac{1}{2^{4k}}\left[\left(\dfrac{(2k)!}{k!k!}\right)^2\right.\\
  &\left.-\dfrac{(m+n+2k)!^2}{2^{2(m+n)}k!(m+k)!(n+k)!(m+n+k)!}\right]
\end{align*}





\section{$J_n(B)$ 的完整推导}

\subsection{从实积分到围道积分}

计算
\begin{equation}
    J_n(B) = \int_{-\pi}^{\pi}\frac{\cos(n\theta)}{B + \cos\theta} \di\theta,
    \label{eq:A1}
\end{equation}
其中 $B$ 是实参数，$n$ 是非负整数。当 $|B|\le 1$ 时分母有实零点，积分理解为主值。

利用Euler公式 $\cos\theta = \frac{e^{i\theta} + e^{-i\theta}}{2}$，作变量代换 $z = e^{i\theta}$。当 $\theta$ 从 $-\pi$ 变到 $\pi$ 时，$z$ 沿单位圆周 $C$（$|z|=1$）逆时针绕行一周。在此代换下：
\begin{equation}
    \cos\theta = \frac{z+z^{-1}}{2},\qquad \cos(n\theta) = \frac{z^n+z^{-n}}{2},\qquad \di\theta = \frac{\di z}{iz}.
\end{equation}

代入 $J_n(B)$：
\begin{align}
    J_n(B) &= \oint_C \frac{z^n+z^{-n}}{2}\cdot \frac{2}{2B+z+z^{-1}}\cdot \frac{\di z}{iz} \nonumber\\
    &= \frac{1}{i}\oint_C \frac{z^n+z^{-n}}{z^2+2Bz+1} \di z.
    \label{eq:A2}
\end{align}

极点分析：被积函数分裂为两项
\begin{equation}
    \frac{z^n}{z^2+2Bz+1} + \frac{z^{-n}}{z^2+2Bz+1}.
\end{equation}
第一项在 $z=0$ 处解析（$n\ge 0$）。第二项在 $z=0$ 处有 $n$ 阶极点（$n\ge 1$）。分母 $z^2+2Bz+1$ 的零点为
\begin{equation}
    z_{\pm} = -B \pm \sqrt{B^2-1},
    \label{eq:A3}
\end{equation}
且 $z_+ z_- = 1$。

\subsection{情况一：$|B| > 1$}

此时 $z_{\pm}$ 均为实数，一根在单位圆内，一根在圆外。以 $B>1$ 为例：$z_+ = -B+\sqrt{B^2-1} \in (-1,0)$ 在圆内，$z_- < -1$ 在圆外。圆内极点为 $z=0$（$n\ge 1$）和 $z=z_+$。

$z=0$ 处留数：令 $g(z) = 1/(z^2+2Bz+1)$，其 Taylor 展开系数为 $a_k = (-1)^k U_k(B)$，其中 $U_k$ 是第二类 Chebyshev 多项式。因此
\begin{equation}
    \operatorname{Res}\left(\frac{z^{-n}}{z^2+2Bz+1}, z=0\right) = a_{n-1} = (-1)^{n-1}U_{n-1}(B).
\end{equation}

$z=z_+$ 处留数：
\begin{equation}
    \operatorname{Res}(z_+) = \frac{z_+^n+z_+^{-n}}{z_+-z_-} = \frac{z_+^n+z_+^{-n}}{2\sqrt{B^2-1}}.
\end{equation}

由留数定理 $\frac{1}{i}\oint_C f(z)\di z = 2\pi\sum_{\text{圆内}}\operatorname{Res}$：
\begin{equation}
    J_n(B) = 2\pi\left[(-1)^{n-1}U_{n-1}(B) + \frac{z_+^n+z_+^{-n}}{2\sqrt{B^2-1}}\right].
\end{equation}

令 $B=\cosh t$（$t>0$），则 $z_+ = -e^{-t}$，$\sqrt{B^2-1} = \sinh t$。利用恒等式 $U_{n-1}(\cosh t) = \sinh(nt)/\sinh t$（见附录B），可化简为
\begin{equation}
    J_n(B) = \frac{2\pi z_+^n}{\sqrt{B^2-1}}.
\end{equation}

对 $B<-1$ 作类似计算，统一写为
\begin{equation}
    J_n(B) = \frac{2\pi z_{\text{in}}^n}{\sqrt{B^2-1}} \operatorname{sgn}(B),
    \label{eq:A4}
\end{equation}
其中 $z_{\text{in}}$ 是单位圆内的根。

\subsection{情况二：$|B| \le 1$（主值）}

此时 $z_{\pm} = -B \pm i\sqrt{1-B^2}$，满足 $|z_{\pm}|=1$，两根均在单位圆周上。积分需理解为主值。采用推迟格林函数方法，令 $B \to B+i\epsilon$（$\epsilon \to 0^+$），计算后取实部。

令 $\cos\theta_0 = -B$，$\sin\theta_0 = \sqrt{1-B^2}$。在 $\epsilon \to 0^+$ 极限下，$z_+$ 移到圆内，$z_-$ 移到圆外。留数之和为
\begin{equation}
    \operatorname{Res}_{\text{tot}} = (-1)^{n-1}U_{n-1}(B) - i\frac{\cos(n\theta_0)}{\sin\theta_0}.
\end{equation}

推迟格林函数 $J_n^{\text{ret}}(B) = 2\pi[(-1)^{n-1}U_{n-1}(B) - i\cos(n\theta_0)/\sin\theta_0]$。取实部得主值：
\begin{equation}
    J_n^{\text{PV}}(B) = 2\pi(-1)^{n-1}U_{n-1}(B) = 2\pi U_{n-1}(-B),
\end{equation}
其中利用了 Chebyshev 多项式的奇偶性 $U_k(-x) = (-1)^k U_k(x)$。对于 $n=0$，仅有 $z_+$ 留数贡献纯虚部，实部为零，故
\begin{equation}
    J_0(B) = 0.
\end{equation}

综上：
\begin{equation}
    \boxed{
    J_0(B) = 0,\qquad J_n(B) = 2\pi U_{n-1}(-B)\quad (|B|\le 1,\ n\ge 1).
    }
    \label{eq:A5}
\end{equation}

\section{$U_k(\cosh t) = \sinh((k+1)t)/\sinh t$ 的证明}

此恒等式在附录A中用于化简 $J_n(B)$。用数学归纳法证明。

$k=0$：$U_0(\cosh t) = 1 = \sinh t/\sinh t$，成立。

$k=1$：$U_1(\cosh t) = 2\cosh t = \sinh(2t)/\sinh t$，成立。

归纳步：设对 $k-1$ 和 $k$ 成立（$k\ge 1$）。由 Chebyshev 递推 $U_{k+1}(x) = 2xU_k(x) - U_{k-1}(x)$：
\begin{align}
    U_{k+1}(\cosh t) &= 2\cosh t\cdot U_k(\cosh t) - U_{k-1}(\cosh t) \nonumber\\
    &= 2\cosh t\frac{\sinh((k+1)t)}{\sinh t} - \frac{\sinh(kt)}{\sinh t} \nonumber\\
    &= \frac{2\cosh t\sinh((k+1)t) - \sinh(kt)}{\sinh t}.
\end{align}

利用 $2\cosh a\sinh b = \sinh(a+b) + \sinh(b-a)$：
\begin{equation}
    2\cosh t\sinh((k+1)t) = \sinh((k+2)t) + \sinh(kt).
\end{equation}

代回得
\begin{equation}
    U_{k+1}(\cosh t) = \frac{\sinh((k+2)t)}{\sinh t}.
\end{equation}
归纳完成。

\section{$S_K = \int_{-\pi}^{\pi} \frac{\sin(K\theta)}{\sin\theta} \di\theta$ 的计算}

此积分在 $\alpha=1$ 的精确解中使用。

$K$ 为奇数：令 $K=2M+1$（$M\ge 0$）。由 Dirichlet 核：
\begin{equation}
    \frac{\sin((2M+1)\theta)}{\sin\theta} = 1 + 2\sum_{j=1}^{M}\cos(2j\theta).
\end{equation}
每一项 $\cos(2j\theta)$（$j\ge 1$）在 $[-\pi,\pi]$ 上积分为零，仅常数项贡献 $2\pi$。故 $S_{2M+1} = 2\pi$。

$K$ 为偶数：令 $K=2M$（$M\ge 1$）：
\begin{equation}
    \frac{\sin(2M\theta)}{\sin\theta} = 2\sum_{j=1}^{M}\cos((2j-1)\theta).
\end{equation}
每一项频率均为奇数（$\neq 0$），积分为零。故 $S_{2M} = 0$。

综上：
\begin{equation}
    \boxed{
    S_K =
    \begin{cases}
        2\pi, & K \text{为奇数},\\
        0, & K \text{为偶数}.
    \end{cases}
    }
    \label{eq:C1}
\end{equation}

\section{Fourier 展开与正交性的完整推导}

\subsection{$\cos^j\theta$ 展开为余弦的线性组合}

由 Euler 公式和二项式定理：
\begin{align}
    \cos^j\theta &= \left(\frac{e^{i\theta}+e^{-i\theta}}{2}\right)^j
    = \frac{1}{2^j}\sum_{p=0}^j \binom{j}{p} e^{i(j-2p)\theta} \nonumber\\
    &= \frac{1}{2^j}\sum_{p=0}^j \binom{j}{p} \cos((j-2p)\theta),
    \label{eq:D1}
\end{align}
其中利用了 $p$ 和 $j-p$ 两项的虚部对消。展开式中出现的余弦频率为 $|j-2p|$（$p=0,\dots,j$），均 $\le j$。

\subsection{$U_{n-1}(-B)$ 展开为 $\cos k_1$ 的多项式}

由 $B = \alpha(1-\cos k_1) - 1$ 得 $-B = 1-\alpha+\alpha\cos k_1$，是 $\cos k_1$ 的一次函数。$U_{n-1}(x)$ 是 $x$ 的 $n-1$ 次多项式。复合后得到 $\cos k_1$ 的至多 $n-1$ 次多项式：
\begin{equation}
    U_{n-1}(1-\alpha+\alpha\cos k_1) = \sum_{j=0}^{n-1} c_j(\alpha)\cos^j k_1.
    \label{eq:D2}
\end{equation}

\subsection{代入 $I(m,n)$ 并逐项积分}

将式 (D2) 代入 $\alpha < 1$ 时 $I(m,n)$ 的表达式 (\ref{eq:I_alpha_less_1})：
\begin{align}
    I(m,n;\alpha) &= -4\pi\int_{-\pi}^{\pi}\cos(mk_1)\sum_{j=0}^{n-1} c_j(\alpha)\cos^j k_1 \di k_1 \nonumber\\
    &= -4\pi\sum_{j=0}^{n-1} c_j(\alpha)\int_{-\pi}^{\pi}\cos(mk_1)\cos^j k_1 \di k_1.
    \label{eq:D3}
\end{align}

将 (D1) 代入：
\begin{equation}
    I(m,n;\alpha) = -4\pi\sum_{j=0}^{n-1}\frac{c_j(\alpha)}{2^j}
    \sum_{p=0}^j \binom{j}{p}
    \int_{-\pi}^{\pi}\cos(mk_1)\cos((j-2p)k_1)\di k_1.
    \label{eq:D4}
\end{equation}

用积化和差：
\begin{equation}
    \cos(mk_1)\cos((j-2p)k_1) = \frac{1}{2}\left[\cos((m+j-2p)k_1) + \cos((m-j+2p)k_1)\right].
\end{equation}

积分 $\int_{-\pi}^{\pi}\cos(Kk_1)\di k_1$ 当 $K=0$ 时为 $2\pi$，$K\neq 0$ 时为零。

当 $m\ge n$ 时，对所有 $j\le n-1$ 和 $p\in[0,j]$：
\begin{itemize}
    \item 频率 $K_1 = m+j-2p \ge m-j \ge m-(n-1) \ge 1 > 0$；
    \item 频率 $K_2 = m-j+2p \ge m-j \ge 1 > 0$。
\end{itemize}
两项频率均严格非零，每一项积分为零。求和后 $I(m,n;\alpha)=0$。

$\alpha$ 仅通过系数 $c_j(\alpha)$ 影响各项权重，不改变频率值。正交性对每个 $(j,p)$ 独立成立。
